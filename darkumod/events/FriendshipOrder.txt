# Darku Question - Initiate
country_event = {
	id = fso.27
	title = fso.27.t
	desc = fso.27.d
	
	
	fire_only_once = yes
	is_triggered_only = yes
	
	immediate = {
		set_country_flag = darku_crisis_active
	}
	
	option = { # kill
		name = fso.27.a
		hidden_effect = { 
			random_list = {
				30 = { # dead, peetry civil war
					CZE = { country_event = { id = fso.28 } }
				}
				40 = { # fail, civil war major
					CZE = { country_event = { id = fso.29 } }
				}
				30 = { # dead, no war
					CZE = { country_event = { id = fso.30 } }
				}
			}
		}
	}
	option = { # talk
		name = fso.27.b 
		hidden_effect = { 
			random_list = {
				30 = { # succeed, Dark resigns
					CZE = { country_event = { id = fso.31 } }
				}
				40 = { # succeed, Dark is icon
					CZE = { country_event = { id = fso.32 } }
				}
				30 = { # fail, cause civil war with Dark 
					CZE = { country_event = { id = fso.33 } }
				}
			}
		}
	}
}

	desc = fso.71.d
	
	
	fire_only_once = yes
	
	trigger = {
		CZE = { has_country_flag = fso_protects_against_fascism }
		AND = {
			OR = {
				AND = {
					is_neighbor_of = GER
					has_defensive_war_with = GER
				}
				AND = {
					is_neighbor_of = ITA
					ITA = { is_in_faction_with = GER }
					has_defensive_war_with = ITA
				}
			}		
			NOT = {
				has_government = fascism
				has_government = democratic
				has_government = communism
			}
		}
	}

		days = 1
	}
	
	option = { # plea for protection
		name = fso.71.a
		add_ai_strategy = {
			type = alliance
			id = "CZE"
			value = 400
		}
		add_ai_strategy = {
			type = befriend
			id = "CZE"			
			value = 400
		}
	}
}
